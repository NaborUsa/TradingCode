input MAlength = 200;
input LowLength = 15;

plot MA = simplemovingavg(length = MAlength);

plot BuySig = low < lowest(low, LowLength)[1] and close > MA;

BuySig.SetPaintingStrategy(paintingstrategy.boolean_arrow_up);
BuySig.AssignValueColor(color.white);
